def tokenize(text):
    return text.lower().split()

def generate_ngrams(tokens, n):
    return [tuple(tokens[i:i+n]) for i in range(len(tokens) - n + 1)]

def jaccard_similarity(list1, list2):
    set1, set2 = set(list1), set(list2)
    return len(set1 & set2) / len(set1 | set2)


text1 = "Computer Science is great field"
text2 = "Artificial intelligence helps computers think"

tokens1 = tokenize(text1)
tokens2 = tokenize(text2)

bigrams1 = generate_ngrams(tokens1, 2)
bigrams2 = generate_ngrams(tokens2, 2)

print("Bigram similarity:", jaccard_similarity(bigrams1, bigrams2))
