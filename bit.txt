plays = {
    "Anthony and Cleopatra": "Anthony is there, Brutus is Caeser is with Cleopatra mercy worser.",
    "Julius Caesar": "Anthony is there, Brutus is Caeser is but Calpurnia is.",
    "The Tempest": "mercy worser",
    "Hamlet": "Caeser and Brutus are present with mercy and worser.",
    "Othello": "Caeser is present with mercy and worser",
    "Macbeth": "Anthony is there, Caeser, mercy."
}

words = ["Anthony", "Brutus", "Caeser", "Calpurnia", "Cleopatra", "mercy", "worser"]

vector_matrix = []
text_list = [text.lower() for text in plays.values()]

for word in words:
    row = []
    for text in text_list:
        row.append(1 if word.lower() in text else 0)
    vector_matrix.append(row)

vector_dict = {}
for i, word in enumerate(words):
    bitstring = ''.join(str(d) for d in vector_matrix[i])
    vector_dict[word.lower()] = int(bitstring, 2)

condition = input("Enter your condition: ").lower()

expr = condition
for word in words:
    expr = expr.replace(word.lower(), str(vector_dict[word.lower()]))

expr = expr.replace(" and ", " & ").replace(" or ", " | ").replace(" not ", " ~")

result_number = eval(expr)
result_bits = bin(result_number)[2:].zfill(len(plays))

print("\nMatching plays:")
for i, bit in enumerate(result_bits):
    if bit == "1":
        print(list(plays.keys())[i])
