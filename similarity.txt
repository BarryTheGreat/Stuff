import math
from collections import Counter

def process_text(text):
    words = text.lower().split()
    return Counter(words)

def cosine_similarity(doc1, doc2):
    all_words = set(doc1) | set(doc2)

    dot = sum(doc1.get(w, 0) * doc2.get(w, 0) for w in all_words)
    mag1 = math.sqrt(sum(v*v for v in doc1.values()))
    mag2 = math.sqrt(sum(v*v for v in doc2.values()))

    return dot / (mag1 * mag2)


doc1 = process_text("this is a sample document")
doc2 = process_text("this document is a test")

print("Similarity:", cosine_similarity(doc1, doc2))
